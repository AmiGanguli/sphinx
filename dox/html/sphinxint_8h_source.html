156  ESphQueryState Switch ( ESphQueryState eNew )
157  {
158  int64_t tmNow = sphMicroTimer();
159  ESphQueryState eOld = m_eState;
160  m_dSwitches [ eOld ]++;
161  m_tmTotal [ eOld ] += tmNow - m_tmStamp;
162  m_eState = eNew;

164  return eOld;
165  }

174  }

179  Switch ( SPH_QSTATE_TOTAL );

1049 {
1050  // empty vector
1051  if ( !dVec.GetLength() )
1052  return -1;
1053 
1054  // check last semi-span
1055  if ( dVec.Last()<tRef || dVec.Last()==tRef )
1056  return dVec.GetLength()-1;
1057 
1058  // linear search for small vectors
1059  if ( dVec.GetLength()<=iSmallTreshold )
1060  {
1061  for ( int i=0; i<dVec.GetLength()-1; i++ )
1062  if ( ( dVec[i]<tRef || dVec[i]==tRef ) && tRef<dVec[i+1] )
1063  return i;
1064  return -1;
1065  }
1066 
1067  // binary search for longer vectors
1068  const T * pStart = dVec.Begin();
1069  const T * pEnd = &dVec.Last();
1070 
1071  if ( ( pStart[0]<tRef || pStart[0]==tRef ) && tRef<pStart[1] )
1072  return 0;
1073 
1074  if ( ( pEnd[-1]<tRef || pEnd[-1]==tRef ) && tRef<pEnd[0] )
1075  return pEnd-dVec.Begin()-1;
1076 
1077  while ( pEnd-pStart>1 )
1078  {
1079  if ( tRef<*pStart || *pEnd<tRef )
1080  break;
1081  assert ( *pStart<tRef );
1082  assert ( tRef<*pEnd );
1083 
1084  const T * pMid = pStart + (pEnd-pStart)/2;
1085  assert ( pMid+1 < &dVec.Last() );
1086 
1087  if ( ( pMid[0]<tRef || pMid[0]==tRef ) && tRef<pMid[1] )
1088  return pMid - dVec.Begin();
1089

1092  else

1094  }
