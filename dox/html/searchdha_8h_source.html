109 // manages persistent connections to a host
110 // serves a FIFO queue.
111 // I.e. if we have 2 connections to a host, and one task rent the connection,
112 // we will return 1-st socket. And the next rent request will definitely 2-nd socket
113 // whenever 1-st socket already released or not.
114 // (previous code used LIFO strategy)
115 struct PersistentConnectionsPool_t
116 {
117  mutable CSphMutex m_dDataLock;
118 private:
119  bool m_bShutdown; // will cause ReturnConnection to close the socket instead of returning it
120  CSphVector<int> m_dSockets GUARDED_BY ( m_dDataLock );
121  int m_iRit GUARDED_BY ( m_dDataLock ); // pos where we take the next socket to rent.
122  int m_iWit GUARDED_BY ( m_dDataLock ); // pos where we will put returned socket.
123  int m_iFreeWindow GUARDED_BY ( m_dDataLock ); // # of free sockets in the existing ring
124  using CGuard = CSphScopedLock<CSphMutex>;
125 
126  int Step ( int* ) REQUIRES ( m_dDataLock ); // step helper over the ring
127 
128 public:
129  PersistentConnectionsPool_t ();
130  void Init ( int iPoolSize ) EXCLUDES ( m_dDataLock );
131  int RentConnection () EXCLUDES ( m_dDataLock );
132  void ReturnConnection ( int iSocket ) EXCLUDES ( m_dDataLock );
133  void Shutdown () EXCLUDES ( m_dDataLock );
134 };

249 struct AgentDash_t : AgentStats_t
250 {
251  uint64_t m_dHostStats[ehMaxStat];
252  DWORD m_uTimestamp;
253 
254  AgentDash_t()
255  : m_uTimestamp ( 0 )
256  {
257  Reset();
258  }
259 
260  void Reset ()
261  {
262  AgentStats_t::Reset ();
263  for ( int i = 0; i<ehMaxStat; ++i )
264  m_dHostStats[i] = 0;
265  }
266 
267  void Add ( const AgentDash_t& rhs )
268  {
269  AgentStats_t::Add ( rhs );
270 
271  if ( m_dHostStats[ehConnTries] )
272  m_dHostStats[ehAverageMsecs] =
273  ( m_dHostStats[ehAverageMsecs] * m_dHostStats[ehConnTries]
274  + rhs.m_dHostStats[ehAverageMsecs] * rhs.m_dHostStats[ehConnTries] )
275  / ( m_dHostStats[ehConnTries] + rhs.m_dHostStats[ehConnTries] );
276  else
277  m_dHostStats[ehAverageMsecs] = rhs.m_dHostStats[ehAverageMsecs];
278  m_dHostStats[ehMaxMsecs] = Max ( m_dHostStats[ehMaxMsecs], rhs.m_dHostStats[ehMaxMsecs] );
279  m_dHostStats[ehConnTries] += rhs.m_dHostStats[ehConnTries];
280  m_dHostStats[ehTotalMsecs] += rhs.m_dHostStats[ehTotalMsecs];
281  }
282 };
