308 template < typename T > T InputBuffer_c::GetT ()
309 {
310  if ( m_bError || ( m_pCur+sizeof(T) > m_pBuf+m_iLen ) )
311  {
312  SetError ( true );
313  return 0;
314  }
315 
316  T iRes = sphUnalignedRead ( *(T*)m_pCur );
317  m_pCur += sizeof(T);
318  return iRes;
319 }
