91  virtual BYTE ** NextDocument ( CSphString & sError )
92  {
93  if ( !IsDocCacheEmpty() )
94  return CopyDoc();
95 
96  if ( m_dFieldLengths.GetLength()!=T::m_tSchema.m_dFields.GetLength() )
97  {
98  int nFields = T::m_tSchema.m_dFields.GetLength();
99  m_dFieldLengths.Resize ( nFields );
100  m_dFieldHasChinese.Resize ( nFields );
101  }
102 
103  m_iDocStart = 0;
104  const int MAX_INDEX_LEN = 8;
105 
106  m_dDocBuffer.Resize(0);
107 
108  while ( !IsDocCacheFull() && m_dDocBuffer.GetLength() < g_iRLPMaxBatchSize )
109  {
110  BYTE ** pFields = T::NextDocument ( sError );
111  if ( !pFields )
112  break;
113 
114  const int * pFieldLengths = T::GetFieldLengths();
115 
116  int iTotalFieldLen = 0;
117  bool bDocHasChinese = false;
118  for ( int i = 0; i < T::m_tSchema.m_dFields.GetLength(); i++ )
119  {
120  m_dFieldLengths[i] = pFieldLengths[i];
121  m_dFieldHasChinese[i] = sphDetectChinese ( pFields[i], m_dFieldLengths[i] );
122  if ( m_dFieldHasChinese[i] )
123  iTotalFieldLen += PROXY_MARKER_LEN+MAX_INDEX_LEN+m_dFieldLengths[i]+3; // _[marker]_[field_id]_[field text]
124 
125  bDocHasChinese |= m_dFieldHasChinese[i];
126  }
127 
128  // no point in caching the document if its the first document and there's nothing CJK in it
129  if ( IsDocCacheEmpty() && !bDocHasChinese )
130  {
131  T::m_tState.m_dFieldLengths.Resize ( m_dFieldLengths.GetLength() );
132  ARRAY_FOREACH ( i, m_dFieldLengths )
133  T::m_tState.m_dFieldLengths[i] = m_dFieldLengths[i];
134 
135  return pFields;
136  }
137 
138  int iCurDoc;
139  StoredDoc_t * pDoc = PushDoc ( iCurDoc );
140  int nFields = T::m_tSchema.m_dFields.GetLength();
141  CopyDocInfo ( pDoc->m_tDocInfo, T::m_tDocInfo );
142  pDoc->m_dMva = T::m_dMva;
143  pDoc->m_dStrAttrs = T::m_dStrAttrs;
144  pDoc->m_dFields.Resize ( nFields );
145  pDoc->m_dFieldLengths.Resize ( nFields );
146  pDoc->m_dFieldStorage.Resize ( nFields );
147 
148  // fields that don't have chinese text will be stored in those docs
149  // fields that have chinese text will be added later, after a RLP field filter pass
150  for ( int i = 0; i < T::m_tSchema.m_dFields.GetLength(); i++ )
151  if ( !m_dFieldHasChinese[i] )
152  {
153  int iFieldLength = m_dFieldLengths[i];
154  pDoc->m_dFieldStorage[i].Resize ( iFieldLength+1 );
155  BYTE * pStart = pDoc->m_dFieldStorage[i].Begin();
156  memcpy ( pStart, pFields[i], iFieldLength );
157  pDoc->m_dFields[i] = pStart;
158  pDoc->m_dFieldLengths[i] = iFieldLength;
159 
160  // stripper still doesn't know about field lengths, so we need to add a zero
161  pDoc->m_dFields[i][iFieldLength] = '\0';
162  }
163 
164  // document doesnt have any CJK, so no copying/segmenting/etc
165  if ( !bDocHasChinese )
166  continue;
167 
168  int iOldBufferLen = m_dDocBuffer.GetLength();
169  m_dDocBuffer.Resize ( iOldBufferLen+PROXY_MARKER_LEN+MAX_INDEX_LEN+2+iTotalFieldLen );
170  BYTE * pCurDocPtr = &(m_dDocBuffer[iOldBufferLen]);
171 
172  // document start tag
173  COPY_MARKER ( pCurDocPtr, m_pMarkerDocStart );
174 
175  // document id
176  AddNumber ( pCurDocPtr, iCurDoc );
177 
178  // flatten all fields(+markers) to one buffer
179  for ( int i = 0; i < T::m_tSchema.m_dFields.GetLength(); i++ )
180  if ( m_dFieldHasChinese[i] )
181  {
182  COPY_MARKER ( pCurDocPtr, m_pMarkerFieldStart );
183 
184  // field id
185  AddNumber ( pCurDocPtr, i );
186  *pCurDocPtr++ = ' ';
187 
188  int iFieldLen = m_dFieldLengths[i];
189  memcpy ( pCurDocPtr, pFields[i], iFieldLen );
190  pCurDocPtr += iFieldLen;
191  }
192 
193  m_dDocBuffer.Resize ( pCurDocPtr-m_dDocBuffer.Begin() );
194  }
195 
196  if ( IsDocCacheEmpty() )
197  return NULL;
198 
199  assert ( m_pBatchFieldFilter && !T::m_pFieldFilter );
200  int iResultLen = m_pBatchFieldFilter->Apply ( m_dDocBuffer.Begin(), m_dDocBuffer.GetLength(), m_dResult, false );
201  assert ( iResultLen );
202 
203  BYTE * pSegmentedStart = m_dResult.Begin();
204  BYTE * pSegmentedEnd = pSegmentedStart+iResultLen;
205  BYTE * pFieldStart = NULL;
206  StoredDoc_t * pCurDoc = NULL;
207  int iFieldId = -1;
208 
209  while ( pSegmentedStart < pSegmentedEnd )
210  {
211  BYTE * pTmp = pSegmentedStart;
212  int iCode = sphUTF8Decode ( (const BYTE * &)pSegmentedStart );
213 
214  if ( iCode!=PROXY_DOCUMENT_START && iCode!=PROXY_FIELD_START )
215  continue;
216 
217  // we have a segmented field that we need to place into the appropriate cached document
218  if ( pFieldStart )
219  {
220  assert ( pCurDoc && iFieldId!=-1 );
221 
222  pCurDoc->m_dFields[iFieldId] = pFieldStart;
223  pCurDoc->m_dFieldLengths[iFieldId] = pTmp-pFieldStart;
224 
225  // stripper still doesn't know about field lengths, so we need to add a zero
226  pTmp[-1] = '\0';
227  }
228 
229  if ( iCode==PROXY_DOCUMENT_START )
230  {
231  // fetch document number
232  int iDoc = ReadNumber ( pSegmentedStart, pSegmentedEnd );
233  pCurDoc = &(m_dBatchedDocs[iDoc]);
234  iFieldId = -1;
235  pFieldStart = NULL;
236 
237  } else if ( iCode==PROXY_FIELD_START )
238  {
239  // fetch field id
240  iFieldId = ReadNumber ( pSegmentedStart, pSegmentedEnd );
241  pFieldStart = pSegmentedStart;
242  }
243  }
244 
245  // last field
246  if ( pFieldStart )
247  {
248  assert ( pCurDoc && iFieldId!=-1 );
249 
250  pCurDoc->m_dFields[iFieldId] = pFieldStart;
251  pCurDoc->m_dFieldLengths[iFieldId] = pSegmentedEnd-pFieldStart;
252  }
253 
254  return CopyDoc ();
255  }

331  void AddNumber ( BYTE * & pPtr, int iNumber )
332  {
333  char szTmp [256];
334  int iLen = snprintf ( szTmp, sizeof(szTmp), "%d", iNumber );
335  iLen = iLen>=0 ? iLen : sizeof(szTmp);
336  memcpy ( pPtr, szTmp, iLen );
337  pPtr += iLen;
338  }

340  int ReadNumber ( BYTE * & pPtr, BYTE * pEndPtr )
341  {
342  int iCode = sphUTF8Decode ( (const BYTE * &)pPtr );
343  assert ( iCode==' ' );
344  const BYTE * pNumber = pPtr;
345  do
346  {
347  iCode = sphUTF8Decode ( (const BYTE * &)pPtr );
348  } while ( pPtr < pEndPtr && iCode!=' ');
349 
350  assert ( iCode==' ' );
351  BYTE uTmp = *pPtr;
352  *pPtr = '\0';
353  int iDoc = atoi ( (char *)pNumber );
354  *pPtr = uTmp;
355 
356  return iDoc;
357  }
