68 inline DWORD sphJsonUnpackInt ( const BYTE ** pp )
69 {
70  const BYTE * p = *pp;
71  DWORD uRes = p[0];
72  switch ( p[0] )
73  {
74  default:
75  (*pp) += 1;
76  break;
77  case 252:
78  uRes = p[1] + ( p[2]<<8 );
79  (*pp) += 3;
80  break;
81  case 253:
82  uRes = p[1] + ( p[2]<<8 ) + ( p[3]<<16 );
83  (*pp) += 4;
84  break;
85  case 254:
86  uRes = p[1] + ( p[2]<<8 ) + ( p[3]<<16 ) + ( p[4]<<24 );
87  (*pp) += 5;
88  break;
89  case 255:
90  assert ( 0 && "unexpected reserved length code 255" );
91  (*pp) += 1;
92  break;
93  }
